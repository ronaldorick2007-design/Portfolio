function linearSearch(arr, target){
    N = arr.length;
    for(let i = 0; i < N; i++ ){
      if(arr[i] == target){
        return i;
      }
    return -1;
    }
  }
A = [1, 4, 2, 5, 3]
linearSearch(A, 2);
